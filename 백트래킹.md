#include <iostream>
#include<vector>
using namespace std;

void backtracking(vector<int>&V, vector<bool>&visit,vector<int>&path, int M)
{
	if (path.size() == M)
	{
		for (int num : path)
			cout << num << ' ';

		cout << '\n'; return;
	}
	for (int i = 0; i < V.size(); ++i)
	{
		if (visit[i] == true)
			continue;

		visit[i] = true;
		path.emplace_back(V[i]);

		backtracking(V, visit, path, M);

		path.pop_back();
		visit[i] = false;
	}
}

int main()
{
	int N, M; cin >> N >> M;
	vector<int>v(N);
	vector<bool>visit(N,false);
	vector<int>path;
	for (int i = 0; i < N; ++i)
		v[i] = i+1;
	backtracking(v, visit,path, M);
}
