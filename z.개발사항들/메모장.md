\begin{algorithm}
\caption{Aggregate Initialization Style Checker}
\begin{algorithmic}
  \Procedure{CheckArrayInit}{$ILE, arrayType$}
    \State $declaredSize \gets \text{getConstantArraySize}(arrayType)$
    \State $explicitCount \gets \text{countInits}(ILE)$
    
    \If{the initializer is zero shorthand form}
      \State \Return
    \EndIf
    
    \If{$explicitCount \neq declaredSize$}
      \State $\text{reportError}(\text{partial initialization not allowed})$
    \EndIf
    
    \State $elemType \gets \text{getElementType}(arrayType)$
    \If{element type is aggregate}
      \For{$i \gets 0$ \To $explicitCount - 1$}
        \State $child \gets \text{getInit}(ILE, i)$
        \If{child is not brace enclosed}
          \State $\text{reportError}(\text{nested braces required})$
        \EndIf
      \EndFor
    \EndIf
  \EndProcedure

  \Procedure{CheckRecordInit}{$ILE, recordType$}
    \State $fieldCount \gets \text{countNonStaticFields}(recordType)$
    \State $explicitCount \gets \text{countInits}(ILE)$
    
    \If{the initializer is zero shorthand form}
      \State \Return
    \EndIf
    
    \If{$explicitCount \neq fieldCount$}
      \State $\text{reportError}(\text{all fields must be specified})$
    \EndIf
  \EndProcedure

  \Procedure{IsZeroShorthand}{$ILE$}
    \State $count \gets \text{getNumInits}(ILE)$
    
    \If{$count = 0$}
      \State \Return $\text{AllowEmptyBraces}$
    \EndIf
    
    \If{$count = 1$}
      \State $elem \gets \text{getInit}(ILE, 0)$
      \State \Return $\text{isZeroLiteral}(elem)$
    \EndIf
    
    \State \Return $\text{false}$
  \EndProcedure

  \Procedure{Check}{$matchResult$}
    \State $ILE \gets \text{getInitListExpr}(matchResult)$
    \State $varDecl \gets \text{getVarDecl}(matchResult)$
    
    \If{location should be filtered out}
      \State \Return
    \EndIf
    
    \State $type \gets \text{getType}(varDecl)$
    
    \If{type is array}
      \State $\text{CheckArrayInit}(ILE, type)$
    \EndIf
    
    \If{type is aggregate record}
      \State $\text{CheckRecordInit}(ILE, type)$
    \EndIf
  \EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Always Constant Condition Detection}
\begin{algorithmic}
  \Procedure{AnalyzeFunctionBody}{$FunctionDecl$}
    \State $EnvStack \gets \text{empty stack}$
    \State $\text{pushGlobalEnv}()$
    \State $\text{traverseFunctionBody}(FunctionDecl)$
    \State $\text{popEnv}()$
  \EndProcedure

  \Procedure{AnalyzeConditionNode}{$Cond, Location$}
    \If{location should not be reported}
      \State \Return
    \EndIf
    
    \If{constant fold phase enabled}
      \If{condition evaluates to compile-time constant}
        \State $\text{reportError}(\text{always constant})$
        \State \Return
      \EndIf
    \EndIf
    
    \If{type range phase enabled}
      \If{condition is always true or false by type range}
        \State $\text{reportError}(\text{always constant by type})$
        \State \Return
      \EndIf
    \EndIf
    
    \If{path constraint phase not enabled}
      \State \Return
    \EndIf
    
    \State $\text{analyzeWithPathConstraints}(Cond)$
  \EndProcedure

  \Procedure{AnalyzeWithPathConstraints}{$Cond$}
    \If{condition is not variable comparison with constant}
      \State \Return
    \EndIf
    
    \State $Var, Op, C \gets \text{decompose}(Cond)$
    \State $TypeInterval \gets \text{getTypeRange}(Var)$
    \State $CurrentInterval \gets \text{getCurrentConstraint}(Var, TypeInterval)$
    
    \If{condition always evaluates same under current interval}
      \State $\text{reportError}(\text{always constant by path constraint})$
    \EndIf
  \EndProcedure

  \Procedure{TraverseIfStmt}{$IfStmt$}
    \State $\text{analyzeConditionNode}(IfStmt.condition)$
    
    \If{condition is simple variable comparison}
      \State $Var, Op, C \gets \text{decompose}(IfStmt.condition)$
      
      \State $ThenEnv \gets \text{narrowIntervalForThen}(Var, Op, C)$
      \State $\text{pushEnv}(ThenEnv)$
      \State $\text{traverseStmt}(IfStmt.thenBranch)$
      \State $\text{popEnv}()$
      
      \If{else branch exists}
        \State $ElseEnv \gets \text{narrowIntervalForElse}(Var, Op, C)$
        \State $\text{pushEnv}(ElseEnv)$
        \State $\text{traverseStmt}(IfStmt.elseBranch)$
        \State $\text{popEnv}()$
      \EndIf
    \EndIf
  \EndProcedure

  \Procedure{NarrowIntervalForThen}{$CurrentInterval, Op, C$}
    \State $L, U \gets CurrentInterval$
    
    \If{$Op$ is less than}
      \State $U \gets \min(U, C - 1)$
    \EndIf
    \If{$Op$ is less than or equal}
      \State $U \gets \min(U, C)$
    \EndIf
    \If{$Op$ is greater than}
      \State $L \gets \max(L, C + 1)$
    \EndIf
    \If{$Op$ is greater than or equal}
      \State $L \gets \max(L, C)$
    \EndIf
    \If{$Op$ is equal}
      \State $L, U \gets C, C$
    \EndIf
    
    \State \Return $\text{Interval}(L, U)$
  \EndProcedure
\end{algorithmic}
\end{algorithm}
