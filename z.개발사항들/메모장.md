```pseudo
\begin{algorithm}
\caption{Macro Defined Before Use Check}
\begin{algorithmic}

  \Procedure{RegisterPPCallbacks}{$SM, PP$}
    \State \texttt{// Entry point, registers callbacks with the preprocessor}
    \State $Callbacks \gets \text{new MacroPPCallbacks(}SM, PP, \text{this)}$
    \State $PP.\text{addPPCallbacks(}Callbacks\text{)}$
  \EndProcedure}

  \Procedure{OnIfDirective}{$Loc, ConditionRange$}
    \State \texttt{// Hook for \#if directives}
    \State AnalyseCondition($ConditionRange$)
  \EndProcedure}

  \Procedure{OnElifDirective}{$Loc, ConditionRange$}
    \State \texttt{// Hook for \#elif directives}
    \State AnalyseCondition($ConditionRange$)
  \EndProcedure}
  
  \Procedure{AnalyseCondition}{$CondRange$}
    \State \texttt{// Analyzes the expression in an \#if or \#elif}
    \If{$CondRange$ is invalid}
      \State \Return
    \EndIf

    \State $SpellingLoc \gets \text{GetSpellingLocation(}CondRange.\text{getBegin())}$
    \If{option 'OnlyMainFile' is \textbf{true} AND NOT $\text{IsInMainFile(}SpellingLoc\text{)}$}
      \State \Return
    \EndIf
    \If{option 'IgnoreSystemHeaders' is \textbf{true} AND $\text{IsInSystemHeader(}SpellingLoc\text{)}$}
      \State \Return
    \EndIf

    \State $Buffer \gets \text{GetSourceText(}CondRange\text{)}$
    \If{$Buffer$ is empty}
      \State \Return
    \EndIf

    \State $Lexer \gets \text{new Lexer(}Buffer\text{)}$
    \State $Tracker \gets \text{new DefinedTracker()}$
    
    \While{$Lexer.\text{LexFromRawLexer(}Tok\text{)}$ is NOT finished}
      \If{$Tok.\text{is(}\text{raw\_identifier}\text{)}$}
        \State $II \gets \text{GetIdentifierInfo(}Tok.\text{getRawIdentifier())}$
        \If{$II.\text{getName()} == \text{"defined"}$}
          \State $Tracker.\text{foundDefined()}$
          \State \textbf{continue}
        \EndIf
        
        \If{$Tracker.\text{shouldSkipMacro()}$}
          \State $Tracker.\text{reset()}$
          \State \textbf{continue}
        \EndIf
        
        \State CheckMacroDefinition($II, Tok$)

      \ElseIf{$Tok.\text{is(}\text{l\_paren}\text{)}$}
        \State $Tracker.\text{foundLeftParen()}$
      \ElseIf{$Tok.\text{is(}\text{r\_paren}\text{)}$}
        \State $Tracker.\text{foundRightParen()}$
      \Else
        \State $Tracker.\text{reset()}$
      \EndIf
    \EndWhile
  \EndProcedure}

  \Procedure{CheckMacroDefinition}{$II, Tok$}
    \State $Name \gets II.\text{getName()}$
    \If{option 'IgnoreBuiltinMacros' is \textbf{true} AND IsBuiltinMacro($Name$)}
      \State \Return
    \EndIf

    \State $status \gets \text{GetMacroDefinitionStatus(}II\text{)}$
    \If{$status == \text{UNDEFINED}$ AND option 'WarnOnUndefined' is \textbf{true}}
        \State \text{ReportDiagnostic(}Tok, \text{"Macro '"} + Name + \text{"' is not defined")}
    \EndIf
  \EndProcedure}

  \Procedure{GetMacroDefinitionStatus}{$II$}
    \If{NOT $\text{IsMacroDefined(}II\text{)}$}
      \State \Return UNDEFINED
    \EndIf
    
    \State $MI \gets \text{GetMacroInfo(}II\text{)}$
    \If{$MI$ is null OR $MI.\text{getNumTokens()} == 0$}
        \State \Return DEFINED\_EMPTY
    \EndIf
    
    \State \Return DEFINED\_WITH\_VALUE
  \EndProcedure}
  
  \Procedure{IsBuiltinMacro}{$Name$}
     \State \texttt{// Checks if macro is built-in like \_\_FILE\_\_}
     \State $PredefinedList \gets$ \{\text{"\_\_FILE\_\_"}, \text{"\_\_LINE\_\_"}, \text{"\_WIN32"}, ...\}
     \If{$Name$ is in $PredefinedList$}
        \State \Return \textbf{true}
     \Else
        \State \Return \textbf{false}
     \EndIf
  \EndProcedure}

\end{algorithmic}
\end{algorithm}
```
