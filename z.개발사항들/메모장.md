```pseudo
\begin{algorithm}
\caption{IncludeCharsetCheck: Header Path Validation}
\begin{algorithmic}

\Procedure{Constructor}{Name, Context}
  \State Load all configuration options from Context
  \State $OnlyMainFile \gets \text{option or default true}$
  \State $IgnoreSystemHeaders \gets \text{option or default true}$
  \State $AllowHyphen \gets \text{option or default false}$
  \State $AllowUnicode \gets \text{option or default false}$
  \State $AllowSpaces \gets \text{option or default false}$
  \State $AllowParentPath \gets \text{option or default false}$
  \State $AllowDoubleSep \gets \text{option or default false}$
  \State $AllowAbsolutePath \gets \text{option or default false}$
  \State $MaxPathLength \gets \text{option or default 260}$
  \State $EnforceForwardSlashOnly \gets \text{option or default true}$
\EndProcedure

\Procedure{ShouldReportAt}{Location, SourceManager, OnlyMain, IgnoreSys}
  \If{Location is not valid}
    \Return false
  \EndIf
  \State $SpellingLoc \gets \text{get spelling location from SourceManager}$
  \If{OnlyMain is true and SpellingLoc is not in main file}
    \Return false
  \EndIf
  \If{IgnoreSys is true and SpellingLoc is in system header}
    \Return false
  \EndIf
  \Return true
\EndProcedure

\Procedure{ExtractRawToken}{SourceManager, LangOptions, Range, Fallback, IsAngled}
  \State Create empty RawHeaderToken structure
  \State $RawToken.IsAngled \gets IsAngled$
  \State $Spelled \gets \text{get source text from Range}$
  \If{Spelled is not empty}
    \State $RawToken.Spelled \gets Spelled$
    \If{Spelled starts and ends with quotes or angle brackets}
      \State $RawToken.Inner \gets \text{remove first and last character}$
    \EndIf
  \EndIf
  \If{Spelled is empty}
    \State $RawToken.Spelled \gets Fallback$
    \State $RawToken.Inner \gets Fallback$
  \EndIf
  \Return RawToken
\EndProcedure

\Procedure{IsAsciiOnly}{String}
  \For{each character in String}
    \If{character value is greater than 127}
      \Return false
    \EndIf
  \EndFor
  \Return true
\EndProcedure

\Procedure{HasWhitespace}{String}
  \For{each character in String}
    \If{character is space tab newline or other whitespace}
      \Return true
    \EndIf
  \EndFor
  \Return false
\EndProcedure

\Procedure{HasForbiddenChars}{String, AllowHyphen, Offenders}
  \State Create seen array of size 256 initialized to false
  \State $hasBadChar \gets false$
  \For{each character in String}
    \If{character is not alphanumeric and not allowed special character}
      \If{character is hyphen and AllowHyphen is true}
        \State Skip this character
      \EndIf
      \If{character has not been seen before}
        \State Add character to Offenders list
        \State Mark character as seen
        \State $hasBadChar \gets true$
      \EndIf
    \EndIf
  \EndFor
  \Return hasBadChar
\EndProcedure

\Procedure{ContainsParentPath}{Path}
  \State Create normalized copy of Path
  \For{each character in normalized path}
    \If{character is backslash}
      \State Replace with forward slash
    \EndIf
  \EndFor
  \If{path equals two dots}
    \Return true
  \EndIf
  \If{path starts with two dots and slash}
    \Return true
  \EndIf
  \If{path ends with slash and two dots}
    \Return true
  \EndIf
  \If{path contains slash two dots slash pattern}
    \Return true
  \EndIf
  \Return false
\EndProcedure

\Procedure{ContainsDoubleSeparator}{Path}
  \If{Path contains double forward slash}
    \Return true
  \EndIf
  \If{Path contains double backslash}
    \Return true
  \EndIf
  \Return false
\EndProcedure

\Procedure{IsAbsolutePath}{Path}
  \If{Path starts with forward slash}
    \Return true
  \EndIf
  \If{Path starts with double backslash}
    \Return true
  \EndIf
  \If{Path has drive letter pattern with colon and separator}
    \Return true
  \EndIf
  \Return false
\EndProcedure

\Procedure{HandleInclusionDirective}{HashLocation, FileName, IsAngled, Range}
  \If{location should not be reported}
    \Return
  \EndIf
  \State $Raw \gets$ ExtractRawToken(SourceManager, LangOptions, Range, FileName, IsAngled)
  \If{$Raw.Inner$ length is greater than MaxPathLength}
    \State Report error for path too long
  \EndIf
  \If{EnforceForwardSlashOnly is true and Raw contains backslash}
    \State Report error for backslash usage
  \EndIf
  \If{AllowUnicode is false and Raw contains non-ASCII characters}
    \State Report error for non-ASCII characters
  \EndIf
  \If{AllowSpaces is false}
    \If{Raw inner path has whitespace}
      \State Report error for whitespace in path
    \EndIf
    \If{Raw spelled token has whitespace}
      \State Report error for whitespace in token
    \EndIf
  \EndIf
  \If{Raw contains comment patterns}
    \State Report error for comment pattern in header name
  \EndIf
  \State Create empty offenders list
  \If{Raw has forbidden characters}
    \State Build detail string with all offending characters
    \State Report error with forbidden character details
  \EndIf
  \If{AllowParentPath is false and Raw contains parent path reference}
    \State Report error for parent directory reference
  \EndIf
  \If{AllowDoubleSep is false and Raw contains double separator}
    \State Report error for consecutive separators
  \EndIf
  \If{AllowAbsolutePath is false and Raw is absolute path}
    \State Report error for absolute path usage
  \EndIf
  \If{Raw spelled token is not empty}
    \State $FirstChar \gets \text{first character of Raw.Spelled}$
    \State $LastChar \gets \text{last character of Raw.Spelled}$
    \If{delimiter pair is not properly matched}
      \State Report error for mismatched delimiters
    \EndIf
  \EndIf
\EndProcedure

\Procedure{RegisterPPCallbacks}{SourceManager, Preprocessor}
  \If{Preprocessor is null}
    \Return
  \EndIf
  \State Create new IncludePPCallbacks instance
  \State Add callbacks to Preprocessor
\EndProcedure

\end{algorithmic}
\end{algorithm}
```
