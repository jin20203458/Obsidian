\begin{algorithm}
\caption{Include Directive Charset Validation}
\begin{algorithmic}
  \Procedure{HandleInclusionDirective}{$HashLoc, FileName, IsAngled, FilenameRange$}
    \If{location should not be reported}
      \State \Return
    \EndIf
    
    \State $Raw \gets \text{extractRawToken}(FilenameRange, FileName, IsAngled)$
    
    \State $\text{validatePathLength}(Raw)$
    \State $\text{validateSlashConsistency}(Raw)$
    \State $\text{validateCharacterSet}(Raw)$
    \State $\text{validatePathStructure}(Raw)$
    \State $\text{validateDelimiterPair}(Raw)$
  \EndProcedure

  \Procedure{ValidateCharacterSet}{$Raw$}
    \If{unicode not allowed and path contains non-ASCII}
      \State $\text{reportError}(\text{non-ASCII characters})$
    \EndIf
    
    \If{spaces not allowed and path contains whitespace}
      \State $\text{reportError}(\text{whitespace characters})$
    \EndIf
    
    \If{path contains comment patterns}
      \State $\text{reportError}(\text{comment patterns})$
    \EndIf
    
    \State $offenders \gets \text{empty list}$
    \If{path has forbidden characters}
      \For{$c$ in path}
        \If{$c$ is not alphanumeric and not allowed symbol}
          \State $\text{add } c \text{ to } offenders$
        \EndIf
      \EndFor
      \State $\text{reportError}(offenders)$
    \EndIf
  \EndProcedure

  \Procedure{ValidatePathStructure}{$Raw$}
    \If{parent path not allowed and path contains dot-dot}
      \State $\text{reportError}(\text{parent directory reference})$
    \EndIf
    
    \If{double separator not allowed and path has consecutive slashes}
      \State $\text{reportError}(\text{double separator})$
    \EndIf
    
    \If{absolute path not allowed and path is absolute}
      \State $\text{reportError}(\text{absolute path})$
    \EndIf
  \EndProcedure
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Macro Definition Before Use Checker}
\begin{algorithmic}
  \Procedure{AnalyseCondition}{$conditionRange$}
    \State $buffer \gets \text{extractSourceText}(conditionRange)$
    \State $lexer \gets \text{createLexer}(buffer)$
    \State $tracker \gets \text{new DefinedTracker}()$
    
    \While{lexer has more tokens}
      \State $token \gets \text{lexer.nextToken}()$
      
      \If{token is identifier}
        \If{token is keyword defined}
          \State $tracker.\text{foundDefined}()$
        \EndIf
        
        \If{tracker should skip this macro}
          \State $tracker.\text{reset}()$
        \EndIf
        
        \If{tracker should not skip}
          \State $\text{checkMacroDefinition}(token)$
        \EndIf
      \EndIf
      
      \If{token is left parenthesis}
        \State $tracker.\text{foundLeftParen}()$
      \EndIf
      
      \If{token is right parenthesis}
        \State $tracker.\text{foundRightParen}()$
      \EndIf
    \EndWhile
  \EndProcedure

  \Procedure{CheckMacroDefinition}{$identifier, token$}
    \State $name \gets identifier.\text{getName}()$
    
    \If{name is builtin macro}
      \State \Return
    \EndIf
    
    \State $status \gets \text{getMacroStatus}(identifier)$
    
    \If{$status = \text{UNDEFINED}$}
      \State $\text{reportError}(token, name)$
    \EndIf
  \EndProcedure

  \Procedure{GetMacroStatus}{$identifier$}
    \If{identifier is not defined}
      \State \Return $\text{UNDEFINED}$
    \EndIf
    
    \State $macroInfo \gets \text{getMacroInfo}(identifier)$
    
    \If{macroInfo has no tokens}
      \State \Return $\text{DEFINED\_EMPTY}$
    \EndIf
    
    \State \Return $\text{DEFINED\_WITH\_VALUE}$
  \EndProcedure
\end{algorithmic}
\end{algorithm}
