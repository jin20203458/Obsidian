\begin{algorithm}
\caption{Include Charset Validation}
\begin{algorithmic}

  \State \textbf{Function Call Order:}
  \State 1. RegisterPPCallbacks (Entry point)
  \State 2. $\to$ InclusionDirective (Preprocessor callback)
  \State 3. $\to$ Handle (Core validation)
  \State 4. $\to$ ExtractRawToken (Extract path)
  \State 5. $\to$ HasForbiddenChars (Char validation)
  \State 6. $\to$ (Other validation helpers)

\State $\phantom{x}$
\State $\phantom{x}$
  
  
  \Function{RegisterPPCallbacks}{$SM, PP$}
    \If{$PP$ is not null}
      \State attach IncludePPCallbacks to $PP$
    \EndIf
  \EndFunction

  \Function{InclusionDirective}{$HashLoc, IncludeTok, FileName, IsAngled, FilenameRange, File, SearchPath, RelativePath$}
    \State call Handle($HashLoc, FileName, IsAngled, FilenameRange$)
  \EndFunction

  \Function{Handle}{$HashLoc, FileName, IsAngled, FilenameRange$}
    \State $canReport \gets \text{ShouldReportAt}(FilenameRange.\text{Begin}, OnlyMainFile, IgnoreSystemHeaders)$
    \If{$canReport$ is false}
      \State \Return
    \EndIf
    
    \State $Raw \gets \text{ExtractRawToken}(FilenameRange, FileName, IsAngled)$
    
    \If{$Raw.Inner.\text{length} > MaxPathLength$}
      \State $\text{reportError}(\text{path exceeds maximum length})$
    \EndIf
    
    \If{$EnforceForwardSlashOnly$ and $Raw.Inner$ contains backslash}
      \State $\text{reportError}(\text{use forward slash only})$
    \EndIf
    
    \If{not $AllowUnicode$ and $Raw.Inner$ is not ASCII only}
      \State $\text{reportError}(\text{non-ASCII characters forbidden})$
    \EndIf
    
    \If{not $AllowSpaces$}
      \If{$Raw.Inner$ contains whitespace}
        \State $\text{reportError}(\text{whitespace in path forbidden})$
      \EndIf
      \If{$Raw.Spelled$ contains whitespace}
        \State $\text{reportError}(\text{whitespace in token forbidden})$
      \EndIf
    \EndIf
    
    \If{$Raw.Spelled$ contains comment pattern}
      \State $\text{reportError}(\text{comment patterns forbidden})$
    \EndIf
    
    \State $offenders \gets \text{empty list}$
    \If{$\text{HasForbiddenChars}(Raw.Inner, AllowHyphen, offenders)$}
      \State $\text{reportError}(\text{forbidden characters detected}, offenders)$
    \EndIf
    
    \If{not $AllowParentPath$ and $Raw.Inner$ contains parent path reference}
      \State $\text{reportError}(\text{parent path reference forbidden})$
    \EndIf
    
    \If{not $AllowDoubleSep$ and $Raw.Inner$ contains double separator}
      \State $\text{reportError}(\text{double separator forbidden})$
    \EndIf
    
    \If{not $AllowAbsolutePath$ and $Raw.Inner$ is absolute path}
      \State $\text{reportError}(\text{absolute path forbidden})$
    \EndIf
    
    \If{$Raw.Spelled$ is not empty}
      \If{$Raw.Spelled$ delimiters are mismatched}
        \State $\text{reportError}(\text{mismatched delimiters})$
      \EndIf
    \EndIf
  \EndFunction

  \Function{ShouldReportAt}{$Loc, OnlyMain, IgnoreSys$}
    \If{$Loc$ is invalid}
      \State \Return false
    \EndIf
    \State $Spell \gets \text{getSpellingLocation}(Loc)$
    \If{$OnlyMain$ and $Spell$ not in main file}
      \State \Return false
    \EndIf
    \If{$IgnoreSys$ and $Spell$ in system header}
      \State \Return false
    \EndIf
    \State \Return true
  \EndFunction

  \Function{ExtractRawToken}{$Range, FallbackName, IsAngled$}
    \State $R.\text{IsAngled} \gets IsAngled$
    \State $Spelled \gets \text{getSourceText}(Range)$
    \If{$Spelled$ is not empty}
      \State $R.\text{Spelled} \gets Spelled$
      \If{$Spelled$ has matching quote or angle bracket delimiters}
        \State $R.\text{Inner} \gets Spelled$ without first and last character
      \Else
        \State $R.\text{Inner} \gets Spelled$
      \EndIf
    \Else
      \State $R.\text{Spelled} \gets FallbackName$
      \State $R.\text{Inner} \gets FallbackName$
    \EndIf
    \State \Return $R$
  \EndFunction

  \Function{HasForbiddenChars}{$S, AllowHyphen, offenders$}
    \State $hasBadChar \gets false$
    \State $seen \gets \text{new boolean array}(256)$
    \For{each character $c$ in $S$}
      \State $allowed \gets false$
      \If{$c$ is alphanumeric}
        \State $allowed \gets true$
      \EndIf
      \If{$c$ is underscore or period or slash or backslash}
        \State $allowed \gets true$
      \EndIf
      \If{$c$ is hyphen and $AllowHyphen$}
        \State $allowed \gets true$
      \EndIf
      \If{not $allowed$}
        \State $hasBadChar \gets true$
        \If{not $seen[c]$}
          \State add $c$ to offenders
          \State $seen[c] \gets true$
        \EndIf
      \EndIf
    \EndFor
    \State \Return $hasBadChar$
  \EndFunction

\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Include Charset Validation}
\begin{algorithmic}

  \Function{registerPPCallbacks}{$SM, PP$}
    \If{$PP$ is not null}
      \State attach IncludePPCallbacks to $PP$
    \EndIf
  \EndFunction

  \Function{InclusionDirective}{$HashLoc, IncludeTok, FileName, IsAngled, FilenameRange, File, SearchPath, RelativePath, SuggestedModule, ModuleImported, FileType$}
    \State call $\text{handle}(HashLoc, FileName, IsAngled, FilenameRange)$
  \EndFunction

  \Function{handle}{$HashLoc, FileName, IsAngled, FilenameRange$}
    \If{location should not be reported}
      \State \Return
    \EndIf
    
    \State $Raw \gets \text{extractRawToken}(FilenameRange, FileName, IsAngled)$
    
    \If{$|Raw.Inner| > MaxPathLength$}
      \State $\text{reportError}(\text{path exceeds maximum length})$
    \EndIf
    
    \If{$EnforceForwardSlashOnly$ and $Raw.Inner$ contains backslash}
      \State $\text{reportError}(\text{use forward slash instead of backslash})$
    \EndIf
    
    \If{not $AllowUnicode$ and $Raw.Inner$ contains non-ASCII}
      \State $\text{reportError}(\text{non-ASCII characters not allowed})$
    \EndIf
    
    \If{not $AllowSpaces$ and ($Raw.Inner$ or $Raw.Spelled$ has whitespace)}
      \State $\text{reportError}(\text{whitespace not allowed})$
    \EndIf
    
    \If{$Raw.Spelled$ contains comment patterns}
      \State $\text{reportError}(\text{comment patterns not allowed})$
    \EndIf
    
    \State $offenders \gets \text{forbidden characters in } Raw.Inner$
    \If{$offenders$ is not empty}
      \State $\text{reportError}(\text{forbidden characters found}, offenders)$
    \EndIf
    
    \If{not $AllowParentPath$ and $Raw.Inner$ contains parent reference}
      \State $\text{reportError}(\text{parent path reference not allowed})$
    \EndIf
    
    \If{not $AllowDoubleSep$ and $Raw.Inner$ contains double separator}
      \State $\text{reportError}(\text{double separator not allowed})$
    \EndIf
    
    \If{not $AllowAbsolutePath$ and $Raw.Inner$ is absolute}
      \State $\text{reportError}(\text{absolute path not allowed})$
    \EndIf
    
    \If{$Raw.Spelled$ bracket pair is mismatched}
      \State $\text{reportError}(\text{mismatched bracket delimiters})$
    \EndIf
  \EndFunction

  \Function{extractRawToken}{$SM, LangOptions, FilenameRange, FallbackName, IsAngled$}
    \State $Spelled \gets \text{getSourceText}(FilenameRange, SM, LangOptions)$
    \If{$Spelled$ is not empty}
      \If{$Spelled$ enclosed in quotes or angle brackets}
        \State $Inner \gets Spelled$ without first and last character
      \Else
        \State $Inner \gets Spelled$
      \EndIf
    \Else
      \State $Spelled \gets FallbackName$
      \State $Inner \gets FallbackName$
    \EndIf
    \State \Return $(Spelled, Inner, IsAngled)$
  \EndFunction

  \Function{hasForbiddenChars}{$S, allowHyphen$}
    \State $offenders \gets \text{empty list}$
    \For{each character $c$ in $S$}
      \If{$c$ not in allowed set for $allowHyphen$}
        \If{$c$ not already recorded}
          \State add $c$ to offenders
        \EndIf
      \EndIf
    \EndFor
    \State \Return offenders
  \EndFunction

  \Function{containsParentPath}{$Path$}
    \State normalize backslashes to forward slashes in $Path$
    \If{$Path$ is exactly ".." or starts with "../" or ends with "/.." or contains "/.."}
      \State \Return true
    \EndIf
    \State \Return false
  \EndFunction

\end{algorithmic}
\end{algorithm}
