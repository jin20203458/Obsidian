using System;
using System.ComponentModel;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;
using PathIO = System.IO.Path;  //  Path 모호성 해결

namespace SentinelStatic.Models
{

    /// Clang-Tidy가 출력하는 분석 결과(진단 메시지) 하나를 나타내는 순수한 데이터 클래스입니다.
    public class DiagnosticItem
    {
        public string FilePath { get; set; } = string.Empty;
        public int Line { get; set; }
        public int Column { get; set; }
        public string Message { get; set; } = string.Empty;
        public string CheckName { get; set; } = string.Empty;
    }

    public class DiagnosticItemEx : DiagnosticItem, INotifyPropertyChanged
    {
        // ============ INotifyPropertyChanged 구현 ============

        public event PropertyChangedEventHandler? PropertyChanged;

        //
        protected void OnPropertyChanged([CallerMemberName] string? propertyName = null)
        {
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(propertyName));
        }

        // 필드를 최신화 하고, 값 변경이 있다면 이벤트 발생
        protected bool SetProperty<T>(ref T field, T value, [CallerMemberName] string? propertyName = null)
        {
            if (Equals(field, value)) return false;
            field = value;
            OnPropertyChanged(propertyName);
            return true;
        }

        // ============ Computed Properties ============
        public string FileName => string.IsNullOrEmpty(FilePath) ? "" : PathIO.GetFileName(FilePath);

        private string _codeSnippet = "";
        public string CodeSnippet
        {
            get => _codeSnippet;
            set
            {
                if (SetProperty(ref _codeSnippet, value))
                {
                    OnPropertyChanged(nameof(HasCodeSnippet));
                }
            }
        }

        public string FilePathWithLine => string.IsNullOrEmpty(FilePath) ? "" : $"{FileName}:{Line}";

        public bool HasCodeSnippet => !string.IsNullOrEmpty(CodeSnippet);

        // ============ AI 분석 관련 속성 ============

        // AI 분석 결과 텍스트
        private string _aiAnalysis = "";
        public string AIAnalysis
        {
            get => _aiAnalysis;
            set
            {
                if (SetProperty(ref _aiAnalysis, value))
                {
                    OnPropertyChanged(nameof(HasAIAnalysis));
                    OnPropertyChanged(nameof(HasNoAIAnalysis));
                    OnPropertyChanged(nameof(AIStatusText));
                }
            }
        }

        // AI 분석 완료 여부
        private bool _isAIAnalyzed = false;
        public bool IsAIAnalyzed
        {
            get => _isAIAnalyzed;
            set
            {
                if (SetProperty(ref _isAIAnalyzed, value))
                {
                    OnPropertyChanged(nameof(AIStatusText));
                }
            }
        }

        // AI 분석 결과 존재 여부
        public bool HasAIAnalysis => !string.IsNullOrEmpty(AIAnalysis);
        public bool HasNoAIAnalysis => !HasAIAnalysis;

        public string AIStatusText => IsAIAnalyzed
            ? (HasAIAnalysis ? "분석 완료" : "분석 실패")
            : "미분석";

        public DateTime? AIAnalysisStartTime { get; set; }
        public DateTime? AIAnalysisEndTime { get; set; }

        public long? AIAnalysisDuration
        {
            get
            {
                if (AIAnalysisStartTime.HasValue && AIAnalysisEndTime.HasValue)
                {
                    return (long)(AIAnalysisEndTime.Value - AIAnalysisStartTime.Value).TotalMilliseconds;
                }
                return null;
            }
        }

        public string? AIAnalysisError { get; set; }

        // ============ 생성자 ============

        public DiagnosticItemEx() { }

        public DiagnosticItemEx(DiagnosticItem source)
        {
            if (source == null)
                throw new ArgumentNullException(nameof(source));

            FilePath = source.FilePath;
            Line = source.Line;
            Column = source.Column;
            Message = source.Message;
            CheckName = source.CheckName;
        }

        // ============ AI 분석 관련 메서드 ============

        public void StartAIAnalysis()
        {
            AIAnalysisStartTime = DateTime.Now;
            IsAIAnalyzed = false;
            AIAnalysis = "";
            AIAnalysisError = null;
        }

        public void CompleteAIAnalysis(string analysis)
        {
            AIAnalysisEndTime = DateTime.Now;
            AIAnalysis = analysis ?? "";
            IsAIAnalyzed = true;
            AIAnalysisError = null;
        }

        public void FailAIAnalysis(string error)
        {
            AIAnalysisEndTime = DateTime.Now;
            AIAnalysis = $"AI 분석 실패:\n{error}";
            IsAIAnalyzed = true;
            AIAnalysisError = error;
        }

        public void ResetAIAnalysis()
        {
            AIAnalysis = "";
            IsAIAnalyzed = false;
            AIAnalysisError = null;
            AIAnalysisStartTime = null;
            AIAnalysisEndTime = null;
        }

        // ============ 유틸리티 메서드 ============

        public override string ToString()
        {
            var sb = new StringBuilder();
            sb.AppendLine($"File: {FilePathWithLine}");
            sb.AppendLine($"Checker: {CheckName}");
            sb.AppendLine($"Message: {Message}");

            if (HasCodeSnippet)
            {
                sb.AppendLine($"Code Snippet Length: {CodeSnippet.Length} chars");
            }

            if (IsAIAnalyzed)
            {
                sb.AppendLine($"AI Status: {AIStatusText}");
                if (AIAnalysisDuration.HasValue)
                {
                    sb.AppendLine($"AI Duration: {AIAnalysisDuration}ms");
                }
            }

            return sb.ToString();
        }

        public DiagnosticItemEx Clone()
        {
            return new DiagnosticItemEx
            {
                FilePath = this.FilePath,
                Line = this.Line,
                Column = this.Column,
                Message = this.Message,
                CheckName = this.CheckName,
                CodeSnippet = this.CodeSnippet,
                AIAnalysis = this.AIAnalysis,
                IsAIAnalyzed = this.IsAIAnalyzed,
                AIAnalysisError = this.AIAnalysisError,
                AIAnalysisStartTime = this.AIAnalysisStartTime,
                AIAnalysisEndTime = this.AIAnalysisEndTime
            };
        }
    }
}
