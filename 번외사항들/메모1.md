using System;
using System.Windows;
using CommunityToolkit.Mvvm.ComponentModel;

using SentinelStatic.Models;

namespace SentinelStatic.ViewModels
{
    public partial class AIAnalysisViewModel : ObservableObject
    {
        private readonly IAIAnalysisService? _aiAnalysisService;
        private CancellationTokenSource? _analysisCts;
        [ObservableProperty]
        private DiagnosticItemEx _diagnostic;

        public bool HasCodeSnippet => !string.IsNullOrEmpty(Diagnostic?.CodeSnippet);

        public string AnalysisDuration
        {
            get
            {
                if (Diagnostic?.AIResult.Duration.HasValue == true)
                {
                    return $"{Diagnostic.AIResult.Duration.Value}ms";
                }
                return "N/A";
            }
        }

        public AIAnalysisViewModel(DiagnosticItemEx diagnostic, IAIAnalysisService? aiAnalysisService)
        {
            _diagnostic = diagnostic ?? throw new ArgumentNullException(nameof(diagnostic));
            _aiAnalysisService = aiAnalysisService;
        }

        [RelayCommand]
        private void CopyToClipboard()
        {
            try
            {
                string content = $@"
📄 파일: {Diagnostic.FilePathWithLine}
🔎 체커: {Diagnostic.CheckName}
⚠️ 메시지: {Diagnostic.Message}

📝 문제 코드:
{Diagnostic.CodeSnippet}

💡 AI 분석 및 수정 제안:
{Diagnostic.AIResult.Analysis}
";
                Clipboard.SetText(content);
                MessageBox.Show("클립보드에 복사되었습니다!", "성공",
                    MessageBoxButton.OK, MessageBoxImage.Information);
            }
            catch (Exception ex)
            {
                MessageBox.Show($"복사 실패: {ex.Message}", "오류",
                    MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }

        [RelayCommand]
        private async Task ReanalyzeAsync()
        {
            if (_aiAnalysisService == null)
            {
                MessageBox.Show("AI 서비스가 초기화되지 않았습니다.", "오류",
                    MessageBoxButton.OK, MessageBoxImage.Warning);
                return;
            }

            try
            {
                Diagnostic.AIResult.Start();

                string result = await _aiAnalysisService.AnalyzeDiagnosticAsync(
                    Diagnostic,
                    Diagnostic.CodeSnippet,
                    ""
                );

                Diagnostic.AIResult.Complete(result);

                OnPropertyChanged(nameof(AnalysisDuration));

                MessageBox.Show("재분석이 완료되었습니다!", "성공",
                    MessageBoxButton.OK, MessageBoxImage.Information);
            }
            catch (Exception ex)
            {
                Diagnostic.AIResult.Fail(ex.Message);
                MessageBox.Show($"재분석 실패: {ex.Message}", "오류",
                    MessageBoxButton.OK, MessageBoxImage.Error);
            }
        }
    }
}

using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using Microsoft.Win32;
using SentinelStatic.Models;
using SentinelStatic.Service;

namespace SentinelStatic.ViewModels

    public partial class ReportViewModel : ObservableObject
    {
        private readonly CodeSnippetService _codeSnippetService;
        private readonly IReportGenerator _htmlGenerator;
        private readonly IReportGenerator _pdfGenerator;
        private readonly IReportGenerator _excelGenerator;
        private readonly IAIAnalysisService _aiAnalysisService;

        [ObservableProperty] private ObservableCollection<DiagnosticItemEx> diagnostics = new();
        [ObservableProperty] private ObservableCollection<FileStatistic> fileStatistics = new();
        [ObservableProperty] private ObservableCollection<CheckerStatistic> checkerStatistics = new();

        [ObservableProperty] private int totalWarnings;
        [ObservableProperty] private int totalFiles;
        [ObservableProperty] private int totalCheckers;

        // AI 분석 관련 속성
        [ObservableProperty] private DiagnosticItemEx? _selectedDiagnostic;
        [ObservableProperty] private bool _isAnalyzing = false;
        [ObservableProperty] private string _analysisStatus = "진단 항목을 선택하세요";
        [ObservableProperty] private int _analyzedCount = 0;

        private CancellationTokenSource? _analysisCancellationTokenSource;

        // 기본 생성자 (디자인 타임용)
        public ReportViewModel() : this(Enumerable.Empty<DiagnosticItem>(),null) {}

        // 실제 생성자 (의존성 주입)
        public ReportViewModel(
            IEnumerable<DiagnosticItem> diagnosticItems,
            IAIAnalysisService? aiAnalysisService = null)
        {
            _codeSnippetService = new CodeSnippetService();
            _htmlGenerator = new HtmlReportGenerator();
            _pdfGenerator = new PdfReportGenerator();
            _excelGenerator = new ExcelReportGenerator();
            _aiAnalysisService = aiAnalysisService!;

            LoadDiagnostics(diagnosticItems);
            CalculateStatistics();
        }
        [RelayCommand]
        private void OpenAIAnalysisWindow()
        {
            if (SelectedDiagnostic == null)
            {
                MessageBox.Show("진단 항목을 먼저 선택하세요.", "알림",
                    MessageBoxButton.OK, MessageBoxImage.Information);
                return;
            }

            if (!SelectedDiagnostic.AIResult.IsAnalyzed)
            {
                MessageBox.Show("AI 분석이 완료되지 않았습니다.", "알림",
                    MessageBoxButton.OK, MessageBoxImage.Information);
                return;
            }

            var viewModel = new AIAnalysisViewModel(SelectedDiagnostic, _aiAnalysisService);
            var window = new AIAnalysisWindow
            {
                DataContext = viewModel,
                Owner = Application.Current.MainWindow
            };
            window.ShowDialog();
        }
        private void LoadDiagnostics(IEnumerable<DiagnosticItem> diagnosticItems)
        {
            var itemsWithCode = diagnosticItems.Select(d =>
            {
                var itemEx = new DiagnosticItemEx(d);

                if (!string.IsNullOrEmpty(d.FilePath) && d.Line > 0)
                {
                    try
                    {
                        itemEx.CodeSnippet = _codeSnippetService.GetCodeSnippet(
                            d.FilePath, d.Line, d.Column
                        );
                    }
                    catch (FileNotFoundException)
                    {
                        itemEx.CodeSnippet = "[파일을 찾을 수 없습니다]";
                    }
                    catch (UnauthorizedAccessException)
                    {
                        itemEx.CodeSnippet = "[파일 접근 권한이 없습니다]";
                    }
                    catch (IOException ex)
                    {
                        itemEx.CodeSnippet = $"[파일 읽기 오류: {ex.Message}]";
                    }
                    catch (Exception ex)
                    {
                        // 예상치 못한 오류 로깅
                        System.Diagnostics.Debug.WriteLine($"Unexpected error: {ex}");
                        itemEx.CodeSnippet = $"[오류: {ex.GetType().Name}]";
                    }
                }

                return itemEx;
            }).ToList();

            Diagnostics = new ObservableCollection<DiagnosticItemEx>(itemsWithCode);
        }


        private void CalculateStatistics()
        {
            var validDiagnostics = Diagnostics.Where(d => !string.IsNullOrEmpty(d.FilePath)).ToList();
            TotalWarnings = validDiagnostics.Count;

            var fileGroups = validDiagnostics
                .GroupBy(d => d.FilePath)
                .Select(g => new FileStatistic
                {
                    FileName = Path.GetFileName(g.Key),
                    WarningCount = g.Count()
                })
                .OrderByDescending(f => f.WarningCount);

            FileStatistics = new ObservableCollection<FileStatistic>(fileGroups);
            TotalFiles = FileStatistics.Count;

            var checkerGroups = validDiagnostics
                .Where(d => !string.IsNullOrEmpty(d.CheckName))
                .GroupBy(d => d.CheckName)
                .Select(g => new CheckerStatistic
                {
                    CheckerName = g.Key,
                    WarningCount = g.Count()
                })
                .OrderByDescending(c => c.WarningCount);

            CheckerStatistics = new ObservableCollection<CheckerStatistic>(checkerGroups);
            TotalCheckers = CheckerStatistics.Count;
        }

        // 선택된 진단 항목 변경 시 자동 AI 분석
        partial void OnSelectedDiagnosticChanged(DiagnosticItemEx? value)
        {
            // ✅ UI 스레드에서 Command 갱신
            Application.Current.Dispatcher.Invoke(() =>
            {
                ReanalyzeCommand?.NotifyCanExecuteChanged();
                OpenAIAnalysisWindowCommand?.NotifyCanExecuteChanged();
            });

            if (value == null)
            {
                AnalysisStatus = "진단 항목을 선택하세요";
                return;
            }

            if (value.AIResult.IsAnalyzed)
            {
                AnalysisStatus = "AI 분석 완료";
                return;
            }

            if (_aiAnalysisService != null)
            {
                _ = AnalyzeWithAIAsync(value);
            }
        }

        private async Task AnalyzeWithAIAsync(DiagnosticItemEx diagnostic, CancellationToken cancellationToken = default)
        {
    if (_aiAnalysisService == null)
    {
        // ✅ UI 스레드에서 실행
        Application.Current.Dispatcher.Invoke(() =>
        {
            diagnostic.AIResult.Analysis = "AI 서비스가 초기화되지 않았습니다.";
        });
        return;
    }

            try
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    IsAnalyzing = true;
                    AnalysisStatus = "AI 분석 중...";
                });

                string aiAnalysis = await _aiAnalysisService.AnalyzeDiagnosticAsync(
                    diagnostic,
                    diagnostic.CodeSnippet,
                    GetRuleContextForChecker(diagnostic.CheckName)
                );

                Application.Current.Dispatcher.Invoke(() =>
                {
                    diagnostic.AIResult.Analysis = aiAnalysis;
                    diagnostic.AIResult.IsAnalyzed = true;
                    AnalyzedCount = Diagnostics.Count(d => d.AIResult.IsAnalyzed);
                    AnalysisStatus = $"AI 분석 완료 ({AnalyzedCount}/{TotalWarnings})";

                    // ✅ 분석 완료 후 Command 갱신
                    AnalyzeAllCommand?.NotifyCanExecuteChanged();
                });
            }
            finally
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    IsAnalyzing = false;
                    // OnIsAnalyzingChanged에서 자동으로 Command 갱신됨
                });
            }
        }


        // 수동 AI 분석 재실행 커맨드
        [RelayCommand(CanExecute = nameof(CanReanalyze))]
        private async Task ReanalyzeAsync()
        {
            if (SelectedDiagnostic == null) return;

            SelectedDiagnostic.AIResult.IsAnalyzed = false;
            SelectedDiagnostic.AIResult.Analysis = "";

            await AnalyzeWithAIAsync(SelectedDiagnostic);
        }

        private bool CanReanalyze()
        {
            return SelectedDiagnostic != null && !IsAnalyzing && _aiAnalysisService != null;
        }

        [RelayCommand(CanExecute = nameof(CanAnalyzeAll))]
        private async Task AnalyzeAllAsync()
        {
            if (_aiAnalysisService == null)
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    MessageBox.Show("AI 서비스가 초기화되지 않았습니다.",
                        "오류", MessageBoxButton.OK, MessageBoxImage.Error);
                });
                return;
            }

            var unanalyzedItems = Diagnostics.Where(d => !d.AIResult.IsAnalyzed).ToList();
            int total = unanalyzedItems.Count;

            if (total == 0)
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    MessageBox.Show("분석할 항목이 없습니다.", "정보",
                        MessageBoxButton.OK, MessageBoxImage.Information);
                });
                return;
            }

            // 사용자 확인 (UI 스레드에서)
            bool confirmed = false;
            Application.Current.Dispatcher.Invoke(() =>
            {
                var result = MessageBox.Show(
                    $"{total}개의 진단 항목을 AI로 분석하시겠습니까?\n" +
                    $"예상 소요 시간: 약 {(total * 2)}초",
                    "일괄 AI 분석",
                    MessageBoxButton.YesNo,
                    MessageBoxImage.Question);
                confirmed = (result == MessageBoxResult.Yes);
            });

            if (!confirmed) return;

            _analysisCancellationTokenSource = new CancellationTokenSource();
            var cancellationToken = _analysisCancellationTokenSource.Token;

            int current = 0;
            int successCount = 0;
            int failureCount = 0;

            try
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    IsAnalyzing = true;
                    AnalyzeAllCommand.NotifyCanExecuteChanged();
                });

                foreach (var diagnostic in unanalyzedItems)
                {
                    if (cancellationToken.IsCancellationRequested) break;

                    current++;

                    Application.Current.Dispatcher.Invoke(() =>
                    {
                        AnalysisStatus = $"AI 분석 중... ({current}/{total}) - " +
                                       $"성공: {successCount}, 실패: {failureCount}";
                    });

                    try
                    {
                        // ✅ CancellationToken 전달
                        await AnalyzeWithAIAsync(diagnostic, cancellationToken);

                        if (diagnostic.AIResult.IsAnalyzed && diagnostic.AIResult.Error == null)
                        {
                            successCount++;
                        }
                        else
                        {
                            failureCount++;
                        }
                    }
                    catch (TaskCanceledException)
                    {
                        // 취소 예외 발생 시 루프 종료
                        break;
                    }
                    catch (OperationCanceledException)
                    {
                        // 취소 예외 발생 시 루프 종료
                        break;
                    }
                    catch (Exception ex)
                    {
                        failureCount++;
                        Application.Current.Dispatcher.Invoke(() =>
                        {
                            diagnostic.AIResult.Fail(ex.Message);
                        });
                    }

                    // Task.Delay도 취소 토큰 사용
                    try
                    {
                        await Task.Delay(500, cancellationToken);
                    }
                    catch (TaskCanceledException)
                    {
                        break;
                    }
                }

                // 취소 여부에 따라 다른 메시지 표시
                if (cancellationToken.IsCancellationRequested)
                {
                    Application.Current.Dispatcher.Invoke(() =>
                    {
                        AnalysisStatus = $"분석 취소됨 - 완료: {successCount + failureCount}/{total}";
                        MessageBox.Show(
                            $"AI 분석이 취소되었습니다.\n\n처리 완료: {successCount + failureCount}개\n" +
                            $"성공: {successCount}개\n실패: {failureCount}개",
                            "분석 취소",
                            MessageBoxButton.OK,
                            MessageBoxImage.Information);
                    });
                }
                else
                {
                    Application.Current.Dispatcher.Invoke(() =>
                    {
                        AnalysisStatus = $"전체 AI 분석 완료 - 성공: {successCount}, 실패: {failureCount}";
                        MessageBox.Show(
                            $"AI 분석이 완료되었습니다.\n\n성공: {successCount}개\n실패: {failureCount}개",
                            "분석 완료",
                            MessageBoxButton.OK,
                            MessageBoxImage.Information);
                    });
                }
            }
            catch (TaskCanceledException)
            {
                // 최상위 취소 예외 처리
                Application.Current.Dispatcher.Invoke(() =>
                {
                    AnalysisStatus = "분석 취소됨";
                });
            }
            finally
            {
                Application.Current.Dispatcher.Invoke(() =>
                {
                    IsAnalyzing = false;
                    AnalyzeAllCommand.NotifyCanExecuteChanged();
                    ReanalyzeCommand.NotifyCanExecuteChanged();
                });

                // CancellationTokenSource 정리
                _analysisCancellationTokenSource?.Dispose();
                _analysisCancellationTokenSource = null;
            }
        }
