 public async Task<string> AnalyzeDiagnosticAsync(
 DiagnosticItem diagnostic,
 string codeSnippet,
 string ruleContext,
 CancellationToken cancellationToken = default)  // ✅ 추가
 {
     if (!_isInitialized)
     {
         throw new InvalidOperationException("AI 서비스가 초기화되지 않았습니다.");
     }

     // ✅ 취소 확인
     cancellationToken.ThrowIfCancellationRequested();

     try
     {
         var prompt = new StringBuilder();
         prompt.AppendLine($"다음 코드에서 '{diagnostic.CheckName}' 규칙 위반이 발견되었습니다:");
         prompt.AppendLine();
         prompt.AppendLine("**코드 스니펫:**");
         prompt.AppendLine("```c or c++");
         prompt.AppendLine(codeSnippet);
         prompt.AppendLine("```");
         prompt.AppendLine();
         prompt.AppendLine($"**진단 메시지:** {diagnostic.Message}");
         prompt.AppendLine($"**파일:** {diagnostic.FilePath}");
         prompt.AppendLine($"**위치:** Line {diagnostic.Line}, Column {diagnostic.Column}");

         if (!string.IsNullOrEmpty(ruleContext))
         {
             prompt.AppendLine();
             prompt.AppendLine($"**규칙 정보:** {ruleContext}");
         }

         prompt.AppendLine();
         prompt.AppendLine("위 코드를 분석하고 수정 제안을 제공해주세요.");

         var messages = new List<ChatMessage>(_conversationHistory)
         {
             new UserChatMessage(prompt.ToString())
         };

         //  취소 확인
         cancellationToken.ThrowIfCancellationRequested();


         var responseTask = _chatClient.CompleteChatAsync(messages);

         //  취소 토큰과 함께 대기
         using (cancellationToken.Register(() => { /* 취소 시 아무 작업 안 함 */ }))
         {
             var response = await responseTask;
             var responseText = response.Value.Content[0].Text;
             return responseText;
         }
     }
     catch (OperationCanceledException)
     {
         throw; //  ViewModel에서 처리
     }
     catch (HttpRequestException ex)
     {
         throw new InvalidOperationException(
             "OpenAI API 연결 실패. 인터넷 연결을 확인하세요.",
             ex
         );
     }
     catch (Exception ex)
     {
         Debug.WriteLine($"[AI] 분석 실패: {ex}");
         throw new InvalidOperationException(
             $"AI 분석 실패: {ex.Message}",
             ex
         );
     }
 }
